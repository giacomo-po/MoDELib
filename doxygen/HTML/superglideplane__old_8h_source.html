97  {

99  if(link->sink->sID==this->sID)
100  {// edge ends at this node

103  if(neighbor.outEdge)
104  {// and outEdge of the same patch is connected to this node
105  assert(neighbor.outEdge->prev=nullptr);
106  assert(link->next=nullptr);
107  neighbor.outEdge->prev=link;
108  link->next=neighbor.outEdge;
109  }
110  }
111  else if(link->source->sID==this->sID)
112  {// edge starts at this node
113  assert(neighbor.outEdge==nullptr);
114  neighbor.outEdge=link;
115  if(neighbor.inEdge)
116  {
117  assert(neighbor.inEdge->next==nullptr);
118  assert(link->prev==nullptr);
119  neighbor.inEdge->next=link;
120  link->prev=neighbor.inEdge;
121  }
122  }
123  else
124  {
125  assert(false && "CONNECTING LINK TO NON-INCIDENT NODE");
