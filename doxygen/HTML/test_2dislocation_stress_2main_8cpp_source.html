20 {
23  Eigen::Matrix<double,3,3> stress(Eigen::Matrix<double,3,3>::Zero()); // 3x3 matrix of zeros
24 
25  // From Hirth and Lothe ch 3.4, p76.
26  // Assume unitary b and mu
27 
28  const double& x(P(0));
29  const double& y(P(1));
30  const double x2(x*x);
31  const double y2(y*y);
32  const double R4(std::pow(x2+y2,2));
33 
34  stress(0,0)=-y*(3.0*x2+y2)/R4; // sigma_xx
35  stress(1,1)= y*( x2-y2)/R4; // sigma_yy
36  stress(2,2)=-2.0*Material<Isotropic>::nu*y/(x2+y2); // sigma_zz
37  stress(0,1)= x*( x2-y2)/R4; // sigma_xy
38  stress(1,0)=stress(0,1); // symmetry
39 
40  return stress/(2.0*M_PI*(1.0-Material<Isotropic>::nu));
41 }
