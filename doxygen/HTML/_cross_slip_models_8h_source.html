42  const int& modelID)

44 
45 
46  switch (modelID)
47  {
48  case 0:
49  {// no cross-slip

52  }
53 
54  case 1:
55  {// deterministic cross-slip model based on max glide PK force
56 
57  assert(link.grains().size()==1 && "THERE SHOULD BE ONLY ONE GRAIN FOR AN INTERNAL SEGMENT");
58 
59  // Find slip system on which screw segment has highest glide force
60  const auto& grain(**link.grains().begin());
61  const auto pki=link.pkIntegral();
62  std::map<double,int> ssMap;
63  for(size_t s=0;s<grain.slipSystems().size();++s)
64  {
65  const auto& slipSystem(grain.slipSystems()[s]);
66  //const VectorDim ncs=slipSystem.unitNormal.normalized();
67  if((slipSystem.s.cartesian()-link.burgers()).norm()<FLT_EPSILON)
68  {
69  const double pkGlide=(pki-pki.dot(slipSystem.unitNormal)*slipSystem.unitNormal).norm();
70  ssMap.emplace(pkGlide,s);
71  }
72  }
73 
74  if(ssMap.size())
75  {
76  const auto& crosSlipSystem(grain.slipSystems()[ssMap.rbegin()->second]); // last element in map has highest pkGlide
77  if(crosSlipSystem.unitNormal.cross(link.glidePlaneNormal()).squaredNorm()>FLT_EPSILON)
78  {// cross slip system is different than original system
79  crossSlipDeq.emplace_back(link.source->sID,
80  link.sink->sID,
81  grain.grainID,
82  ssMap.rbegin()->second);
83  }
84  }
85  break;
86  }
87 
88 
89  default:
90  {
91  assert(0 && "CROSS-SLIP MODEL NOT IMPLEMENTED");
92  break;
93  }
