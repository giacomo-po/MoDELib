39  void update(VectorType& H,
40  MatrixType& M,
41  const int& k)
42  {
43  const int n=B.cols();
44 
45  // B1=B;
46  // B1(:,k-1)=B(:,k);
47  // B1(:,k)=B(:,k-1);
48  B.col(k).swap(B.col(k-1));
49 
50  MatrixType H1=H;
51  MatrixType M1=M;
52  H1(k-1)=H(k)+std::pow(M(k,k-1),2)*H(k-1);
53  // M1(k,k-1)=M(k,k-1)*H(k-1)/H1(k-1);
54  M1(k,k-1)*=H(k-1)/H1(k-1);
55  H1(k)=H(k-1)-std::pow(M1(k,k-1),2)*H1(k-1);
56  // for i=k+1:n
57  for(int i=k+1;i<n;++i)
58  {
59  M1(i,k-1)=M(i,k-1)*M1(k,k-1)+M(i,k)*H(k)/H1(k-1);
60  M1(i,k)=M(i,k-1)-M(i,k)*M(k,k-1);
61  }
62  // for j=1:k-2
63  for(int j=0;j<=k-2;++j)
64  {
65  M1(k-1,j)=M(k,j);
66  M1(k,j)=M(k-1,j);
67  }
68  H=H1;
69  M=M1;
70  }
