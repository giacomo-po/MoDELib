123  std::map<std::pair<size_t,size_t>,DislocationSegmentIO<dim>> segments() const
124  {
125 
126  std::map<std::pair<size_t,size_t>,DislocationSegmentIO<dim>> temp;
127 
128  for(const auto& link : links())
129  {
130 
131 
132  const auto loopIter=loopMap().find(link.loopID);
133  assert(loopIter!=loopMap().end());
134 
135  const size_t sourceID(std::min(link.sourceID,link.sinkID));
136  const size_t sinkID(std::max(link.sourceID,link.sinkID));
137  const auto key=std::make_pair(sourceID,sinkID);
138 
139  const auto insertPair=temp.insert(std::make_pair(key,DislocationSegmentIO<dim>(sourceID,sinkID)));
140  const auto& iter=insertPair.first;
141  const bool& success=insertPair.second;
142 
143  if(success)
144  {
145  iter->second.n=loopIter->second->N;
146  }
147  else
148  {
149  if(iter->second.n.cross(loopIter->second->N).norm()>FLT_EPSILON)
150  {
151  iter->second.n.setZero();
152  }
153  }
154 
155  if(link.sourceID<link.sinkID)
156  {
157  iter->second.b+=loopIter->second->B;
158  }
159  else
160  {
161  iter->second.b-=loopIter->second->B;
162  }
163 
164  if(iter->second.meshLocation==-1)
165  {
166  iter->second.meshLocation=link.meshLocation;
167  }
168  else
169  {
170  assert(iter->second.meshLocation==link.meshLocation);
171  }
172 
173  }
174 
175  return temp;
176  }
