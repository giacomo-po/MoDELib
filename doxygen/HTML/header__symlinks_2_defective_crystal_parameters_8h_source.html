47  void manageRestart()
48  {
49  // Menage restart
50  IDreader<'F',1,200,double> vReader;
51  vReader.readLabelsFile("F/F_labels.txt");
52  if (vReader.isGood(0,true))
53  {// F/F_0.txt exists
54  vReader.read(0,true);
55  if(runID<0)
56  {// Restart from last available step
57  if(vReader.size())
58  {// at least a line is available
59 // vReader.readLabelsFile("F/F_labels.txt");
60  runID=vReader.rbegin()->first;
61  totalTime=vReader.last("time [b/cs]");
62  dt=vReader.last("dt [b/cs]");
63  }
64  else
65  {// file is empty, keep default initialization
66  runID=0;
67  }
68  }
69  else
70  {// Restart from specific runID
71 // const auto iter=vReader.find(runID);
72  totalTime=vReader(runID,"time [b/cs]");
73  dt=vReader(runID,"dt [b/cs]");
74  }
75  }
76  else
77  {// F/F_0.txt is not there, keep default initialization
78  model::cout<<"Unable to read F/F_0.txt"<<std::endl;
79  runID=0;
80  }
81 
82  model::cout<<"starting at time step "<<runID<<std::endl;
83  model::cout<<"totalTime= "<<totalTime<<std::endl;
84  model::cout<<"dt= "<<dt<<std::endl;
85  }
