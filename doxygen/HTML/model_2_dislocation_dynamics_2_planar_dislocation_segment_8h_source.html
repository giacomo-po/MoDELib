250  /**********************************************************************/
251  void assembleGlide()
252  {
253  this->updateForcesAndVelocities(*this,quadPerLength,false);
254  Fq= this->quadraturePoints().size()? this->nodalVelocityVector() : VectorNdof::Zero();
255  Kqq=this->nodalVelocityMatrix(*this);
256  h2posMap=this->hermite2posMap();
257  Mseg.setZero(Ncoeff*dim,h2posMap.size()*dim);
258  size_t c=0;
259  for(const auto& pair : h2posMap)
260  {
261  for(int r=0;r<Ncoeff;++r)
262  {
263  Mseg.template block<dim,dim>(r*dim,c*dim)=pair.second.first(r)*MatrixDim::Identity();
264  }
265  c++;
266  }
267  }

269  /**********************************************************************/
270  void addToGlobalAssembly(std::deque<Eigen::Triplet<double> >& kqqT,
271  Eigen::VectorXd& FQ) const
272  {
275  if(!hasZeroBurgers())
276  {
277  const Eigen::MatrixXd tempKqq(Mseg.transpose()*Kqq*Mseg); // Create the temporaty stiffness matrix and push into triplets
278  size_t localI=0;
279  for(const auto& pairI : h2posMap)
280  {
281  for(int dI=0;dI<dim;++dI)
282  {
283  const size_t globalI=pairI.first*dim+dI;
284  size_t localJ=0;
285  for(const auto& pairJ : h2posMap)
286  {
287  for(int dJ=0;dJ<dim;++dJ)
288  {
289  const size_t globalJ=pairJ.first*dim+dJ;
290 
291  if (std::fabs(tempKqq(localI,localJ))>FLT_EPSILON)
292  {
293  kqqT.emplace_back(globalI,globalJ,tempKqq(localI,localJ));
294  }
295  localJ++;
296  }
297  }
298  localI++;
299  }
300  }
301 
302  const Eigen::VectorXd tempFq(Mseg.transpose()*Fq); // Create temporary force vector and add to global FQ
303  localI=0;
304  for(const auto& pairI : h2posMap)
305  {
306  for(int dI=0;dI<dim;++dI)
307  {
308  const size_t globalI=pairI.first*dim+dI;
309 
310  FQ(globalI)+=tempFq(localI);
311 
312  localI++;
313  }
314  }
315  }
316  }
