308  VectorDim snapToBoundingBox(const VectorDim& P) const
309  {
316  VerboseDislocationNode(4,"snapping P="<<P.transpose()<<std::endl;);//<<", lineID="<<pLcontained.second<<std::endl;
317 
318 
319  const VectorDim pL=std::get<0>(boundingBoxSegments().snap(P));
320  const VectorDim pV=boundingBoxSegments().snapToVertex(P).second;
321 
322 
323  bool pLcontained=true;
324  bool pVcontained=true;
325 
326  std::deque<VectorDim> bndChords;
327 
328  for(const auto& pair : this->neighbors())
329  {
330  VerboseDislocationNode(3,"checking "<<std::get<1>(pair.second)->source->sID<<"->"<<std::get<1>(pair.second)->sink->sID<<std::endl;);
331 
332  if(std::get<1>(pair.second)->isBoundarySegment())
333  {// boundary segments must not become internal
334  VerboseDislocationNode(3,std::get<1>(pair.second)->source->sID<<"->"<<std::get<1>(pair.second)->sink->sID<<" is boundary"<<std::endl;);
335 
336  pLcontained*=std::get<1>(pair.second)->boundingBoxSegments().contains(0.5*(pL+std::get<0>(pair.second)->get_P())).first;
337  pVcontained*=std::get<1>(pair.second)->boundingBoxSegments().contains(0.5*(pV+std::get<0>(pair.second)->get_P())).first;
338  VerboseDislocationNode(3,"pLcontained="<<pLcontained<<std::endl;);//<<", lineID="<<pLcontained.second<<std::endl;
339  VerboseDislocationNode(3,"pVcontained="<<pVcontained<<std::endl;);//", lineID="<<pVcontained.second<<std::endl;
340 
341  bndChords.push_back(std::get<1>(pair.second)->chord().normalized());
342  }
343 
344  if(std::get<1>(pair.second)->isGrainBoundarySegment())
345  {// grainBoundary segments must not become internal
346  for(const auto& gb : std::get<1>(pair.second)->grainBoundaries())
347  {
348  pLcontained*=gb->contains(pL);
349  pVcontained*=gb->contains(pV);
350 
351  bndChords.push_back(std::get<1>(pair.second)->chord().normalized());
352  }
353  }
354  }
355 
356  bool parallelBndChords=true;
357  for(size_t i=0;i<bndChords.size();++i)
358  {
359  for(size_t j=i+1;j<bndChords.size();++j)
360  {
361  parallelBndChords*=(bndChords[i].cross(bndChords[j]).norm()<FLT_EPSILON);
362  }
363  }
364 
365  if(pLcontained && parallelBndChords)
366  {
367  VerboseDislocationNode(4,"snapping to pL="<<pL.transpose()<<std::endl;);//<<", lineID="<<pLcontained.second<<std::endl;
368  return pL;
369  }
370  else
371  {
372  if(pVcontained)
373  {
374  VerboseDislocationNode(4,"snapping to pV="<<pV.transpose()<<std::endl;);//<<", lineID="<<pLcontained.second<<std::endl;
375  return pV;
376  }
377  else
378  {
379  model::cout<<"DislocationNode "<<this->sID<<" snapToBoundingBox FAILED."<<std::endl;
380  assert(false && "snapToBoundingBox FAILED.");
381  return VectorDim::Zero();
382  }
383  }

388  const Simplex<dim,dim>* get_includingSimplex(const Simplex<dim,dim>* const guess) const
389  {
390  std::pair<bool,const Simplex<dim,dim>*> temp(false,NULL);
391  if (this->network().use_boundary)
392  {
393  if (guess==NULL)
394  {
395  temp=this->network().mesh.search(this->get_P());
396  }
397  else
398  {
399  if(grains().size()==1)
400  {// node only in one region
401  if((*grains().begin())->grainID!=guess->region->regionID)
402  {
403  temp=this->network().mesh.searchRegion((*grains().begin())->grainID,this->get_P());
404  }
405  else
406  {
407  temp=this->network().mesh.searchRegionWithGuess(this->get_P(),guess);
408  }
409  }
410  else
411  {
412  temp=this->network().mesh.searchWithGuess(this->get_P(),guess);
413  }
414  }
415 
416  if(!temp.first) // DislocationNode not found inside mesh
417  {
418  // Detect if the DislocationNode is sligtly outside the boundary
419  int faceID;
420  const double baryMin(temp.second->pos2bary(this->get_P()).minCoeff(&faceID));
421  const bool isApproxOnBoundary(std::fabs(baryMin)<1.0e3*FLT_EPSILON && temp.second->child(faceID).isBoundarySimplex());
422  if(!isApproxOnBoundary)
423  {

426  model::cout<<"bary "<<temp.second->pos2bary(this->get_P())<<std::endl;
427  model::cout<<"face of barymin is "<<temp.second->child(faceID).xID<<std::endl;
428  model::cout<<"face of barymin is boundary Simplex? "<<temp.second->child(faceID).isBoundarySimplex()<<std::endl;
429  model::cout<<"face of barymin is region-boundary Simplex? "<<temp.second->child(faceID).isRegionBoundarySimplex()<<std::endl;
430  assert(0 && "DISLOCATION NODE OUTSIDE MESH.");
431  }
432  }
433  }
434 
435  return temp.second;
436  }
