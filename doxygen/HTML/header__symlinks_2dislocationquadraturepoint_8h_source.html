124  template<typename LinkType>
125  static VectorDim getGlideVelocity(const LinkType& parentSegment,
126  const VectorDim& r,
127  const VectorDim& fPK,
128  const MatrixDim& S,
129  const VectorDim& rl,
130  const double& dL
131  )
132  {
133  VectorDim n(parentSegment.glidePlaneNormal()); // plane normal
134  VectorDim b(parentSegment.burgers()); // Burgers vector
135  VectorDim t(rl); // tangent vector
136 
137  // Select right-handed normal whenever possible
138  if(parentSegment.loopLinks().size()==1)
139  {// pick right-handed normal for n
140  const typename LinkType::LoopLinkType& loopLink(**parentSegment.loopLinks().begin());
141  if(std::fabs(loopLink.loop()->slippedArea())>FLT_EPSILON)
142  {
143  if(parentSegment.source->sID!=loopLink.source()->sID)
144  {// NetworkLink and LoopLink are oriented in opposite direction
145  b*=-1.0;
146  t*=-1.0;
147  }
148  }
149  }
150 
151  VectorDim glideForce = fPK-fPK.dot(n)*n;
152  double glideForceNorm(glideForce.norm());
153 
154  if(glideForceNorm<FLT_EPSILON && parentSegment.network().use_stochasticForce)
155  {
156  glideForce=parentSegment.chord().cross(n);
157  glideForceNorm=glideForce.norm();
158  if(glideForceNorm>FLT_EPSILON)
159  {
160  glideForce/=glideForceNorm;
161  }
162  }
163 
164  VectorDim vv=VectorDim::Zero();
165  if(glideForceNorm>FLT_EPSILON)
166  {
167  double v =parentSegment.network().poly.mobility->velocity(S,b,t,n,
168  parentSegment.network().poly.T,
169  dL,parentSegment.network().simulationParameters.dt,parentSegment.network().use_stochasticForce);
170  assert((parentSegment.network().use_stochasticForce || v>= 0.0) && "Velocity must be a positive scalar");
171  const bool useNonLinearVelocity=true;
172  if(useNonLinearVelocity && v>FLT_EPSILON)
173  {
174  v= 1.0-std::exp(-v);
175  }
176 
177  for(const auto& inclusion : parentSegment.network().eshelbyInclusions() )
178  {// Add EshelbyInclusions stress
179  if(inclusion.second.contains(r))
180  {
181  v*=inclusion.second.mobilityReduction;
182  }
183  }
184 
185 
186  vv= v * glideForce/glideForceNorm;
187  }
188  return vv;
189  }

283  const double& quadPerLength,
284  const double& isClimbStep)
285  {

288 
289  if(this->size())

291  if(parentSegment.network().computeDDinteractions)
292  {
293  const double L0(parentSegment.chord().norm());
294  const VectorDim c(0.5*(parentSegment.source->get_P()+parentSegment.sink->get_P()));
295  for(const auto& link : parentSegment.network().links())
296  {
297  if( !link.second->hasZeroBurgers()
298  && !(link.second->isBoundarySegment() && parentSegment.network().simulationParameters.simulationType==DefectiveCrystalParameters::FINITE_NO_FEM) // exclude boundary segments even if they are non-zero Burgers

302 
303  const StraightDislocationSegment<dim>& ss(link.second->straight);
304 
305 
306  for(const auto& shift : parentSegment.network().periodicShifts)
307  {
308  SegmentSegmentDistance<dim> ssd(ss.P0,ss.P1,
309  parentSegment.source->get_P()+shift,parentSegment.sink->get_P()+shift);
310 
311  // const double dr(ssd.dMin/(L0+ss.length));
312  const double dr(ssd.dMin/(L0));
313 
314  if(dr<10.0)

320  }
321  else if(dr<100.0)

323  const MatrixDim stressSource(ss.stress(parentSegment.source->get_P()+shift));
324  const MatrixDim stressSink(ss.stress(parentSegment.sink->get_P()+shift));

328  qPoint.stress += (1.0-u)*stressSource+u*stressSink;
329  }
330  }
331  else
332  {// 1pt interpolation
333  const MatrixDim stressC(ss.stress(c+shift));
334  for (auto& qPoint : quadraturePoints())
335  {
336  qPoint.stress += stressC;
337  }
338  }
339 
340  }
341  }
342  }
343  }
344 
345 
346  // Add other stress contributions, and compute PK force
347  for (auto& qPoint : quadraturePoints())
348  {
349 
350  if(parentSegment.network().externalLoadController)
351  {// Add stress of externalLoadController
352  qPoint.stress += parentSegment.network().externalLoadController->stress(qPoint.r);
353  }
354 
355  if(parentSegment.network().bvpSolver)

361  {// Add GB stress

363  }

367  for(const auto& shift : parentSegment.network().periodicShifts)

383  const double& quadPerLength,
384  const bool& isClimbStep)
385  {
386 
387  this->clear();

396  const MatrixNcoeffDim qH(parentSegment.hermiteDofs());

398  {
399  this->emplace_back(parentSegment,q,order,SFCH,qH);

401  }
