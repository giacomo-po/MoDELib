34  static SolutionType findIntersection(const VectorDimD& P0,
35  const VectorDimD& Normal,
36  const VectorDimD& v0,
37  const VectorDimD& v1)
38  {
39  const double nNorm(Normal.norm());
40  assert(nNorm>FLT_EPSILON && "PLANE MUST HAVE NON-ZERO NORMAL");
41  const VectorDimD n(Normal/nNorm);
42 
43  // check intersection of v0->v1 with plane
44  // x=v0+u(v1-v0)
45  // (x-P0).n=0
46  // (v0+u(v1-v0)-P0).n=0
47  // u=(P0-v0).n/(v1-v0).n;
48  const double edgeNorm=(v1-v0).norm();
49  if(edgeNorm<FLT_EPSILON)
50  {
51  if(fabs((P0-0.5*(v0+v1)).dot(n))<FLT_EPSILON)
52  {
53  return std::make_tuple(INCIDENT,v0,v1);
54  }
55  else
56  {
57  return std::make_tuple(OFFSET,VectorDimD::Zero(),VectorDimD::Zero());
58  }
59  }
60  else
61  {
62  const double den=(v1-v0).dot(n);
63  const double num=(P0-v0).dot(n);
64  const double P0v0norm=(P0-v0).norm();
65 
66  const double numCheck= (P0v0norm<FLT_EPSILON)? 0.0 : num/P0v0norm;
67 
68  if (fabs(den/edgeNorm)>FLT_EPSILON)
69  {
70  // edge intersects plane
71  const double u=num/den;
72 
73  if(fabs(u)<FLT_EPSILON)
74  {
75  return std::make_tuple(INCIDENT,v0,v0);
76  }
77  else if (u>=FLT_EPSILON && u<=1.0-FLT_EPSILON)
78  {
79  const VectorDimD x((1.0-u)*v0 + u*v1);
80  return std::make_tuple(INCIDENT,x,x);
81  }
82  else if (fabs(1.0-u)<FLT_EPSILON)
83  {
84  return std::make_tuple(INCIDENT,v0,v0);
85  }
86  else
87  {// no roots
88  return std::make_tuple(OFFSET,VectorDimD::Zero(),VectorDimD::Zero());
89  }
90 
91  }
92  else
93  {
94  if (fabs(numCheck)>FLT_EPSILON)
95  {// edge is parallel to plane, no intersection
96  return std::make_tuple(PARALLEL,VectorDimD::Zero(),VectorDimD::Zero());
97 
98  }
99  else
100  {// edge is coplanar
101  return std::make_tuple(COINCIDENT,v0,v1);
102  }
103  }
104  }
105  }
