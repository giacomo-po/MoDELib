69  void readTN(const int& meshID)
70  {
71  clear();
72  IDreader<'N',1,dim,double> nodeReader;
73  nodeReader.read(meshID,true);
74  for(const auto& node : nodeReader)
75  {
76  nodes().emplace(node.first,Eigen::Map<const Eigen::Matrix<double,dim,1>>(node.second.data()));
77  }
78  assert(nodeReader.size()==nodes().size());
79 
80  IDreader<'T',1,dim+2,size_t> elementReader;
81  elementReader.read(meshID,true);
82  for(const auto& eIter : elementReader)
83  {
84  typename SimplexTraits<dim,dim>::SimplexIDType key;
85  for(int d=0;d<dim+1;++d)
86  {
87  key[d]=eIter.second[d];

89  size_t regionID(eIter.second[dim+1]);
90  elements().emplace(eIter.first,std::make_pair(key,regionID));
91  }
92  assert(elementReader.size()==elements().size());
