46  std::tuple<MatrixDimD,MatrixDimD,MatrixDimD> getLatticeBases(const MatrixDimD& A,const MatrixDimD& Q)
47  {
48 
49  // Check that Q is orthogonal
50  const MatrixDimD QQT(Q*Q.transpose());
51  if((QQT-MatrixDimD::Identity()).norm()>2.0*DBL_EPSILON*dim*dim)
52  {
53  std::cout<<"Q=\n"<<Q<<std::endl;
54  std::cout<<"Q*Q^T=\n"<<QQT<<std::endl;
55  assert(false && "ROTATION MATRIX IS NOT ORTHOGONAL.");
56  }
57 
58  // Check sure that C2G is proper
59  assert(std::fabs(Q.determinant()-1.0) < FLT_EPSILON && "ROTATION MATRIX IS NOT PROPER.");
60 
61  // Check that A is full rank
62  assert(std::fabs(A.determinant())>FLT_EPSILON && "A matrix is singular");
63 
64  const MatrixDimD QA(Q*A);
65  return std::make_tuple(QA,QA.inverse().transpose(),Q);

135  /**********************************************************************/
136  static Eigen::Matrix<long int,dim,1> rationalApproximation(VectorDimD nd)
137  {
138  Eigen::Array<long int,dim,1> nums=Eigen::Matrix<long int,dim,1>::Zero();
139 
140  if(nd.squaredNorm()>0.0)
141  {
142  const Eigen::Array<double,dim,1> nda(nd.array().abs()); // vector of close-to-integer numbers corresponding to lattice coordinates
143  size_t maxID=0;
144  const double maxVal(nda.maxCoeff(&maxID));
145  nd/=maxVal; // make each value of nd in [-1:1]
146 
147  nums=Eigen::Matrix<long int,dim,1>::Ones();
148  Eigen::Array<long int,dim,1> dens=Eigen::Matrix<long int,dim,1>::Ones();
149  long int denProd=1;
150 
151  for(int k=0;k<dim;++k)
152  {
153  BestRationalApproximation bra(nd(k),10000);
154 
155  nums(k)=bra.num;
156  dens(k)=bra.den;
157  denProd*=bra.den;
158  }
159 
160  for(int k=0;k<dim;++k)
161  {
162  nums(k)*=(denProd/dens(k));
163  }
164  }
165 
166  return nums.matrix();
167  }
