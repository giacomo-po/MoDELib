73  {
74  // Read data
75  if(reader().isGood(frameID,false))
76  {
77  reader().read(frameID,false);
78  }
79  else
80  {
81  reader().read(frameID,true);
82  }
83 
84  if(false)
85  {// Color by typeID
86  std::set<int> typeIDs;
87  for(const auto& pk : reader())
88  {
89  Eigen::Map<const Eigen::Matrix<float,1,14>> row(pk.second.data());
90  typeIDs.insert(row(13));
91  }
92 
93  Eigen::Matrix<float,3,1> color00((Eigen::Matrix<float,3,1>()<<38.0,38.0,38.0).finished()/255.0);
94  Eigen::Matrix<float,3,1> color05((Eigen::Matrix<float,3,1>()<<153.0,115.0,0.0).finished()/255.0);
95  Eigen::Matrix<float,3,1> color10((Eigen::Matrix<float,3,1>()<<77.0,0.0,0.0).finished()/255.0);
96 
97  lookUpColors->SetNumberOfTableValues(typeIDs.size());
98  for(int f=0;f<typeIDs.size();++f)
99  {
100  const float u= double(f)/(typeIDs.size()-1);
101 
102  Eigen::Matrix<float,3,1> color=color00;
103  if(u<=0.5)
104  {
105  color=color00*(1.0-u/0.5)+color05*u/0.5;
106  }
107  else
108  {
109  color=color05*(1.0-(u-0.5)/0.5)+color10*(u-0.5)/0.5;
110  }
111  lookUpColors->SetTableValue(f ,color(0) ,color(1) ,color(2) ,0.5); // red
112 
113  }
114 
115  if(typeIDs.size()>1)
116  {
117  mapper->SetScalarRange(0, typeIDs.size()-1); // to scale color label (without, col should be between 0 and 1)
118  }
119  }
120 
121  // insert each datapoint
122  for(const auto& pk : reader())
123  {
124  Eigen::Map<const Eigen::Matrix<float,1,14>> row(pk.second.data());
125  points->InsertNextPoint(row(0), row(1), row(2));
126  diameters->InsertNextValue(2.0*row(3)); // origin of arrow
127  colors->InsertNextValue(row(13));
128 
129  }
130 
131  grid->SetPoints(points);
132  grid->GetPointData()->AddArray(diameters);
133  grid->GetPointData()->SetActiveScalars("diameters"); // to set radius first
134  grid->GetPointData()->AddArray(colors);
135 
136  }
