55  static PlaneMeshIntersectionContainerType reducedPlaneMeshIntersection(const PlaneMeshIntersectionContainerType& mpi)
56  {
57  PlaneMeshIntersectionContainerType temp;
58 
59  size_t tailID=mpi.size()-1;
60  for(size_t k=0;k<mpi.size();++k)
61  {
62  const size_t k1=(k==mpi.size()-1? 0 : k+1);
63  const VectorDim dX=mpi[k].second-mpi[tailID].second;
64  const VectorDim dXnew=mpi[k1].second-mpi[k].second;
65  const double dXnorm=dX.norm();
66  const double dXnewNorm=dXnew.norm();
67  if( dXnorm > FLT_EPSILON
68  && dXnewNorm > FLT_EPSILON
69  && dX.cross(dXnew).norm()>FLT_EPSILON*dXnorm*dXnewNorm)
70  {// non zero angle
71  tailID=k;
72  temp.push_back(mpi[k]);
73  }
74 
75  }
76 // model::cout<<"("<<mpi.size()<<"->"<<temp.size()<<")"<<std::endl;
77  return temp;

179  static SiblingsContainerType validUncles(const Simplex<dim,0>& vertex,
180  const int& rID,
181  const VectorDim& N)
182  {
183 
184  // Collect triangles attached to vertex
185  std::set<const Simplex<dim,dim-1>*> grandParents;
186  for(const auto& parent : vertex.parents())
187  {
188  for(const auto& grandParent : parent->parents())
189  {
190  grandParents.insert(grandParent);
191  }
192  }
193 
194  // Loop over triangles and collect valid edges
195  SiblingsContainerType uncles;
196  for(const auto& grandParent : grandParents)
197  {
198  if( (grandParent->isBoundarySimplex() || grandParent->isRegionBoundarySimplex())
199  && grandParent->isInRegion(rID)

201  {
202  for(int c=0; c<Simplex<dim,dim-1>::nFaces;++c)
203  {
204  uncles.insert(&grandParent->child(c));

347  /**********************************************************************/
348  static RootContainerType planeEdgeIntersection(const VectorDim& P0,
349  const VectorDim& n,
350  const Simplex<dim,dim-2>& edge,
351  std::set<const EdgeSimplexType*>& tested)
352  {
353  // std::deque<VectorDim> temp;
354  tested.insert(&edge);
355 
356  RootContainerType temp;
357 
358  const VectorDim& v0(edge.child(0).P0);
359  const VectorDim& v1(edge.child(1).P0);
360 
361  // check intersection of v0->v1 with plane
362  // x=v0+u(v1-v0)
363  // (x-P0).n=0
364  // (v0+u(v1-v0)-P0).n=0
365  // u=(P0-v0).n/(v1-v0).n;
366  const double edgeNorm=(v1-v0).norm();
367  assert(edgeNorm>FLT_EPSILON && "mesh edge has zero norm.");
368  const double den=(v1-v0).dot(n);
369  const double num=(P0-v0).dot(n);
370  const double P0v0norm=(P0-v0).norm();
371 
372  const double numCheck= (P0v0norm<FLT_EPSILON)? 0.0 : num/P0v0norm;
373 
374  if (fabs(den/edgeNorm)>meshIntersectionTol)
375  {
376  // edge intersects plane
377  const double u=num/den;
378 
379  if(fabs(u)<meshIntersectionTol)
380  {
381  temp.emplace_back(v0,&edge.child(0));
382  }
383  else if (u>=meshIntersectionTol && u<=1.0-meshIntersectionTol)
384  {
385  temp.emplace_back((1.0-u)*v0 + u*v1,nullptr);
386  }
387  else if (fabs(1.0-u)<meshIntersectionTol)
388  {
389  temp.emplace_back(v1,&edge.child(1));
390  }
391  else
392  {// no roots
393 
394  }
395 
396  }
397  else
398  {
399  if (fabs(numCheck)>meshIntersectionTol)
400  {// edge is parallel to plane, no intersection
401 
402  }
403  else
404  {// edge is coplanar
405  temp.emplace_back(v0,&edge.child(0));
406  temp.emplace_back(v1,&edge.child(1));
407  }
408  }
409 
410  return temp;
411  // return std::make_pair(&edge,temp);
412  }
