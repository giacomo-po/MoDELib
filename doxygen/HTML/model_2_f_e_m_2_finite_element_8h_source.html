110  // Insert elements
111  for (const auto& simpl : mesh.simplices())
112  {
113  auto temp=ElementContainerType::emplace(simpl.first,ElementType(simpl.second,*this,*this,mesh2femIDmap()));
114  assert(temp.second && "UNABLE TO INSERT ELEMENT IN ELEMENT CONTAINER.");
115 
116  // Add element pointer to each of its nodes
117  for(int n=0;n<ElementType::nodesPerElement;++n)
118  {
119  auto temp1=temp.first->second.node(n).emplace(&temp.first->second);
120  assert(temp1.second && "UNABLE TO INSERT ELEMENT IN NODE.");

122  }

128  }
129 
130  // Check that node[k].gID==k;
131  for(size_t n=0;n<nodes().size();++n)
132  {
133  assert(node(n).gID==n);
134  }
135 
136  // Compute _xMin and _xMax
137 
138  if(nodeSize())
139  {
140  _xMin=node(0).P0;
141  _xMax=node(0).P0;
142 
143  for (size_t n=0;n<nodeSize();++n)
144  {
145  for(int d=0;d<dim;++d)
146  {
147  if (node(n).P0(d)<_xMin(d))
148  {
149  _xMin(d)=node(n).P0(d);
150  }
151  if (node(n).P0(d)>_xMax(d))
152  {
153  _xMax(d)=node(n).P0(d);
154  }
155  }
156  }
157  _xMean=0.5*(_xMin+_xMax);
158 
159  }
