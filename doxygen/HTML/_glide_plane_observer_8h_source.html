166  /**********************************************************************/
167  const PlanePlaneIntersectionType& glidePlaneIntersection(const MeshPlaneType* const p1,
168  const MeshPlaneType* const p2)
169  {/*@param[in] p1 first plane
170  *@param[in] p2 second plane
171  *\returns the infinite line of interseciton between the two planes,
172  * if already computed. Otherwise it computes the interseciton, stores it,
173  * and returns it.
174  */
175  const auto key=std::make_pair(std::max(p1->sID,p2->sID),std::min(p1->sID,p2->sID));
176  const auto iter=glidePlaneIntersections().find(key);
177  if(iter==glidePlaneIntersections().end())
178  {
179  const bool success=glidePlaneIntersections().emplace(std::piecewise_construct,
180  std::make_tuple(key),
181  std::make_tuple(p1->P,p1->unitNormal,p2->P,p2->unitNormal)
182  ).second;
183  assert(success);
184  }
185  return glidePlaneIntersections().at(key);
186  }
