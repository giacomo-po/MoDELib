123  std::tuple<double,double,double> loopLength() const
124  {
125  double freeLength=0.0;
126  double boundaryLength=0.0;
127  double junctionLength=0.0;
128  for(const auto& link : this->links())
129  {
130  if(link.second->pLink->isBoundarySegment())
131  {
132  boundaryLength+=(link.second->sink()->get_P()-link.second->source()->get_P()).norm();
133  }
134  else
135  {
136  if(link.second->pLink->loopLinks().size()==1)
137  {
138  freeLength+=(link.second->sink()->get_P()-link.second->source()->get_P()).norm();
139  }
140  else
141  {
142  junctionLength+=(link.second->sink()->get_P()-link.second->source()->get_P()).norm();
143  }
144  }
145  }
146  return std::make_tuple(freeLength,junctionLength,boundaryLength);
147  }
