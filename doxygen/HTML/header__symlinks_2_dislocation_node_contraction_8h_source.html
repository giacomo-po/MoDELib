112  bool contract(std::shared_ptr<NodeType> nA,
113  std::shared_ptr<NodeType> nB)
114  {
115 
116  VerboseNodeContraction(1,"DislocationNodeContraction::contract "<<nA->sID<<" "<<nB->sID<<std::endl;);
117 
118  const bool nAisMovable=nA->isMovableTo(nB->get_P());
119  const bool nBisMovable=nB->isMovableTo(nA->get_P());
120 
121  if(nAisMovable && nBisMovable)
122  {
123  VerboseNodeContraction(1,"DislocationNodeContraction case 1a"<<std::endl;);
124  return contractYoungest(nA,nB);
125  }
126  else if(nAisMovable && !nBisMovable)
127  {
128  VerboseNodeContraction(1,"DislocationNodeContraction case 1b"<<std::endl;);
129  return contractSecondAndVirtual(nB,nA);
130  }
131  else if(!nAisMovable && nBisMovable)
132  {
133  VerboseNodeContraction(1,"DislocationNodeContraction case 1c"<<std::endl;);
134  return contractSecondAndVirtual(nA,nB);
135  }
136  else
137  {// nA and nB cannot be moved to each other. The calculation of a third point is necessary
138  VerboseNodeContraction(1,"DislocationNodeContraction case 1d"<<std::endl;);
139 
140  const double maxRange=4.0*(nA->get_P()-nB->get_P()).norm();
141 
142 
143  if(nA->isOnBoundary() || nB->isOnBoundary())
144  {// either one of the nodes is a boundary node. Therefore the contraction point must be a boundary node
145 
146  //BoundingMeshSegments<dim> temp(nA->boundingBoxSegments(),nB->boundingBoxSegments());
147  const ConfinedDislocationObject<dim> cdo(*nA,*nB);
148  const BoundingMeshSegments<dim>& temp(cdo.boundingBoxSegments());
149 
150  VerboseNodeContraction(1,"temp.size="<<temp.size()<<std::endl;);
151 
152  switch (temp.size())
153  {
154 
155  case 0:
156  {// no intersection of the bounding boxes
157  VerboseNodeContraction(1,"DislocationNodeContraction case 1f"<<std::endl;);
158  return false;
159  break;
160  }
161 
162  case 1:
163  {
164  const std::shared_ptr<MeshBoundarySegment<dim>>& mbs(temp.front());
165  if((mbs->P0-mbs->P1).norm()<FLT_EPSILON)
166  {// a unique intersection point of the bounding boxes exist
167  VerboseNodeContraction(1,"DislocationNodeContraction case 1d"<<std::endl;);
168  const VectorDim X=0.5*(mbs->P0+mbs->P1);
169  return contractToPosition(nA,nB,X,maxRange);
170  }
171  else
172  {// two possible intersection points of the bounding boxes exist
173  const bool firstIsCloser=(nA->get_P()-mbs->P0).norm()+(nB->get_P()-mbs->P0).norm()<(nA->get_P()-mbs->P1).norm()+(nB->get_P()-mbs->P1).norm();
174  const VectorDim X= firstIsCloser? mbs->P0 : mbs->P1;
175  const VectorDim Y= firstIsCloser? mbs->P1 : mbs->P0;
176 
177  VerboseNodeContraction(1,"DislocationNodeContraction case 1dX"<<std::endl;);
178  const bool Xcontracted=contractToPosition(nA,nB,X,maxRange);
179  if(Xcontracted)
180  {
181  return true;
182  }
183  else
184  {
185  VerboseNodeContraction(1,"DislocationNodeContraction case 1dY"<<std::endl;);
186  const bool Ycontracted=contractToPosition(nA,nB,Y,maxRange);
187  if(Ycontracted)
188  {
189  return true;
190  }
191  else
192  {
193  return false;
194  }
195  }
196  }
197  break;
198  }
199 
200  default:
201  {// bounding boxes intersect in more than one line
202 
203  std::map<double,VectorDim> vertexMap;
204  for(const auto& seg : temp)
205  {
206  const double firstRange=(nA->get_P()-seg->P0).norm()+(nB->get_P()-seg->P0).norm();
207  if(firstRange<maxRange)
208  {
209  vertexMap.insert(std::make_pair(firstRange,seg->P0));
210  }
211 
212  const double secondRange=(nA->get_P()-seg->P1).norm()+(nB->get_P()-seg->P1).norm();
213  if(secondRange<maxRange)
214  {
215  vertexMap.insert(std::make_pair(secondRange,seg->P1));
216  }
217  }
218 
219  bool success=false;
220  for(const auto& pair : vertexMap)
221  {
222  success=contractToPosition(nA,nB,pair.second,maxRange);
223  if(success)
224  {
225  break;
226  }
227  }
228  return success;
229 
230  break;
231  }
232 
233  }
234 
235  }
236  else
237  {// neither a nor b are on bounding box
238  VerboseNodeContraction(1,"DislocationNodeContraction case 5"<<std::endl;);
239  if(nA->glidePlaneIntersections() && nB->glidePlaneIntersections())
240  {// both nodes confined by more then one plane
241 
242  SegmentSegmentDistance<dim> ssd(nA->glidePlaneIntersections()->P0,nA->glidePlaneIntersections()->P1,
243  nB->glidePlaneIntersections()->P0,nB->glidePlaneIntersections()->P1);
244 
245  const auto iSeg=ssd.intersectionSegment();
246  if(iSeg.size()==1)
247  {// incident intersection
248  VerboseNodeContraction(1,"DislocationNodeContraction case 5a"<<std::endl;);

262  }

272

277  }
278  else if(pli.type==PlaneLineIntersection<dim>::INCIDENT)
279  {// _glidePlaneIntersections becomes a singular point
280  VerboseNodeContraction(1,"DislocationNodeContraction case 6b"<<std::endl;);

283  {// intersection point is inside mesh
284  VerboseNodeContraction(1,"DislocationNodeContraction case 6b1"<<std::endl;);
285  return contractToPosition(nA,nB,pli.P,maxRange);
286  }
287  else
288  {
289  return false;
290  }
291  }
292  else
293  {// parallel planes, cannot contract
294  VerboseNodeContraction(1,"DislocationNodeContraction case 6c"<<std::endl;);
295  return false;
296  }
297  }
298  else if(!nA->glidePlaneIntersections() && nB->glidePlaneIntersections())
299  {// same as previous case, so switch nA and nB
300  VerboseNodeContraction(1,"DislocationNodeContraction case 7a"<<std::endl;);
301  return contract(nB,nA);
302  }
303  else
304  {// both nodes confined by only one plane
305 
306  assert(nA->glidePlanes().size()==1);

308

310 
311  if(ppi.type==PlanePlaneIntersection<dim>::COINCIDENT)
312  {// the contraction point can be the averago of nA and nB, which should be internal for convex domains

314  return contractToPosition(nA,nB,nA->glidePlaneIntersections()->snap(0.5*(nA->get_P()+nB->get_P())),maxRange);
315  }
316  else if(ppi.type==PlanePlaneIntersection<dim>::INCIDENT)
317  {

321 // BoundingMeshSegments<dim> temp(nA->boundingBoxSegments(),nB->boundingBoxSegments());

324  case 2:
