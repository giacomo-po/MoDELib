1 % RGB Rgb triple for given CSS color name
2 %
3 % RGB = RGB('COLORNAME') returns the red-green-blue triple corresponding
4 % to the color named COLORNAME by the CSS3 proposed standard [1], which
5 % contains 139 different colors (an rgb triple is a 1x3 vector of
6 % numbers between 0 and 1). COLORNAME is case insensitive, and for gray
7 % colors both spellings (gray and grey) are allowed.
8 %
9 % RGB CHART creates a figure window showing all the available colors with
10 % their names.
11 %
12 % EXAMPLES
13 % c = rgb('DarkRed') gives c = [0.5430 0 0]
14 % c = rgb('Green') gives c = [0 0.5 0]
15 % plot(x,y,'color',rgb('orange')) plots an orange line through x and y
16 % rgb chart shows all the colors
17 %
18 % BACKGROUND
19 % The color names of [1] have already been ratified in [2], and
20 % according to [3] they are accepted by almost all web browsers and are
21 % used in Microsoft's .net framework. All but four colors agree with
22 % the X11 colornames, as detailed in [4]. Of these the most important
23 % clash is green, defined as [0 0.5 0] by CSS and [0 1 0] by X11. The
24 % definition of green in Matlab matches the X11 definition and gives a
25 % very light green, called lime by CSS (many users of Matlab have
26 % discovered this when trying to color graphs with 'g-'). Note that
27 % cyan and aqua are synonyms as well as magenta and fuchsia.
28 %
29 % ABOUT RGB
30 % This program is public domain and may be distributed freely.
31 % Author: Kristján Jónasson, Dept. of Computer Science, University of
32 % Iceland (jonasson@hi.is). June 2009.
33 %
34 % REFERENCES
35 % [1] "CSS Color module level 3", W3C (World Wide Web Consortium)
36 % working draft 21 July 2008, http://www.w3.org/TR/css3-color
37 %
38 % [2] "Scalable Vector Graphics (SVG) 1.1 specification", W3C
39 % recommendation 14 January 2003, edited in place 30 April 2009,
40 % http://www.w3.org/TR/SVG
41 %
42 % [3] "Web colors", http://en.wikipedia.org/wiki/Web_colors
43 %
44 % [4] "X11 color names" http://en.wikipedia.org/wiki/X11_color_names

46 function rgb = rgb(s)
47  persistent num name
48  if isempty(num) % First time rgb is called
49  [num,name] = getcolors();
50  name = lower(name);
51  num = reshape(hex2dec(num), [], 3);
52  % Divide most numbers by 256 for "aesthetic" reasons (green=[0 0.5 0])
53  I = num < 240; % (interpolate F0--FF linearly from 240/256 to 1.0)
54  num(I) = num(I)/256;
55  num(~I) = ((num(~I) - 240)/15 + 15)/16; + 240;
56  end
57  if strcmpi(s,'chart')
58  showcolors()
59  else
60  k = find(strcmpi(s, name));
61  if isempty(k)
62  error(['Unknown color: ' s]);
63  else
64  rgb = num(k(1), :);
65  end
66  end
67 end

99  y = y - h;
100  for k = n(i):n(i+1)-1
101  c = rgb(name{k});
102  bright = (c(1)+2*c(2)+c(3))/4;
103  if bright < 0.5, txtcolor = 'w'; else txtcolor = 'k'; end
104  rectangle('position',[x+d,y,w-2*d,h],'facecolor',c);
105  t = text(x+w/2, y+h/2, name{k}, 'color', txtcolor);
106  set(t, 'vert', 'mid', 'horiz', 'cent', 'fontsize', 9);
107  y = y - h;
108  end
109  y = y - 0.3*h;
110  end
111  x = x + w;
112  end
113 end
