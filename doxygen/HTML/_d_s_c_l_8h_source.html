68  {
71  const MatrixDimD R(B.covBasis()*A.contraBasis().transpose());
72 
73  // Check that R is a proper rotation
74  const MatrixDimD RRT=R*R.transpose();
75  const double RRTmInorm=(RRT-Eigen::Matrix<double,dim,dim>::Identity()).norm()/Eigen::Matrix<double,dim,dim>::Identity().norm();
76  if(RRTmInorm>FLT_EPSILON)
77  {
78  std::cout<<"R="<<std::endl<<R<<std::endl;
79  std::cout<<"R*R^T="<<std::endl<<RRT<<std::endl;
80  std::cout<<"norm(R*R^T-I)/norm(I)="<<RRTmInorm<<", tol="<<FLT_EPSILON<<std::endl;
81  assert(0 && "R IS NOT ORTHOGONAL.");
82  }
83  // make sure that C2G is proper
84  assert(std::fabs(R.determinant()-1.0) < FLT_EPSILON && "R IS NOT PROPER.");
85 
86 
87  // Compute the transition matrix T=inv(A)*B
88  const MatrixDimD T=A.contraBasis().transpose()*B.covBasis();
89 
90  // For the two lattices to have coincident sites, R must be a rational matrix
91  // Compute the integer matrix P and the integer sigma such that T=P/sigma
92  RationalMatrix<dim> rm(T);
93  const MatrixInt& P=rm.integerMatrix;
94  _sigma=rm.den;
95 
96  // The integer matrix P can be decomposed as P=X*D*Y using the Smith decomposition.
97  // X and Y are unimodular, and D is diagonal with D(k,k) dividing D(k+1,k+1)
98  // The decopmosition also computes the matices U and V such that D=U*P*V
99  SmithDecomposition<dim> sd(P);
100 
101  // From T=inv(A)*B=P/sigma=X*D*Y/sigma=X*D*inv(V)/sigma, we have
102  // B1*(sigma*I)=A1*D
103  // where
104  // B1=B*V
105  // A1=A*X
106  // Since V and X are unimodular matrices, B1 and A1 are new bases
107  // of the lattices B and A, respectively. Moreover, since
108  // (sigma*I) and D are diagonal, the columns of B1 and A1 are
109  // proportional, with rational proportionality factors different for each column.
110  // For example, columns "i" read
111  // b1_i*sigma=a1_i*D(i,i)
112  // Therefore, the i-th primitive vectors of the DSCL is
113  // c_i=b1_i*sigma/gcd(sigma,D(i,i))=a1_i*D(i,i)/gcd(sigma,D(i,i))
114  // or, in matrix form
115  // C=B1*N=A1*M, that is
116  // C=B*V*N=A*X*M
117  // where M=diag(D(i,i)/gcd(sigma,D(i,i))) and
118  // N=diag(sigma/gcd(sigma,D(i,i))) and
119 
120  MatrixInt M(MatrixInt::Identity());
121  MatrixInt N(MatrixInt::Identity());
122  for(int i=0;i<dim;++i)
123  {
124  const IntValueType& dii=sd.matrixD()(i,i);
125  M(i,i)=dii/gcd(_sigma,dii);
126  N(i,i)=_sigma/gcd(_sigma,dii);
127  }
128 
129  const MatrixDimD D1=A.covBasis()*sd.matrixX().template cast<double>()*N.template cast<double>().inverse();
130  const MatrixDimD D2=B.covBasis()*sd.matrixV().template cast<double>()*M.template cast<double>().inverse();
131 
132  assert((D1-D2).norm()<FLT_EPSILON && "DSCL calculation failed.");
133 
134  if(useRLLL)
135  {
136  this->setLatticeBasis(RLLL(0.5*(D1+D2),0.75).reducedBasis());
137  }
138  else
139  {
140  this->setLatticeBasis(0.5*(D1+D2));
141  }
142  }

148  }
