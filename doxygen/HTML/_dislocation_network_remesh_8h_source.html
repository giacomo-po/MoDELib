51  const double vTolcont=0.0;

53  std::pair<bool,double> temp=std::make_pair(false,0.0);
54  const VectorDimD chord(segment.chord()); // this is sink->get_P() - source->get_P()
55  const double chordLength(chord.norm());
56  const VectorDimD dv(segment.sink->get_V()-segment.source->get_V());
57  bool endsAreApproaching( chord.dot(dv) < 0.0 );

59  if (((endsAreApproaching || segment.isBoundarySegment())// ends are approaching
60  && dv.norm()*DN.get_dt()>vTolcont*chordLength // contraction is large enough compared to segment length

63  // || segment.isSimpleBndSegment()
64  //|| segment.isSimpleSessile()
65  )
66  {
67  temp=std::make_pair(true,chordLength);
68  }
