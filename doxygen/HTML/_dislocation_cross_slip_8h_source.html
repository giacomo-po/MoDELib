58  CrossSlipContainerType findCrossSlipSegments(const int& crossSlipModel) const
59  {
60 
61  const double sinCrossSlipRad(std::sin(crossSlipDeg*M_PI/180.0));
62  CrossSlipContainerType crossSlipDeq;
63 
64  for(const auto& link : DN.links())
65  {
66  if( !link.second->isBoundarySegment()
67  && !link.second->isGrainBoundarySegment()
68  && !link.second->isGrainBoundarySegment()
69  && !link.second->hasZeroBurgers()
70  && link.second->isGlissile()
71  && link.second->chord().normalized().cross(link.second->burgers().normalized()).norm()<=sinCrossSlipRad
72  )
73  {
74  const auto& grain(**link.second->grains().begin());
75  const int& materialZ(grain.material());
76  switch (materialZ)
77  {
78  case Grain<dim>::Al.Z:
79  CrossSlipModels<typename PeriodicElement<Grain<dim>::Al.Z,Isotropic>::CrystalStructure>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
80  break;
81  case Grain<dim>::Ni.Z:
82  CrossSlipModels<typename PeriodicElement<Grain<dim>::Ni.Z,Isotropic>::CrystalStructure>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
83  break;
84  case Grain<dim>::Cu.Z:
85  CrossSlipModels<typename PeriodicElement<Grain<dim>::Cu.Z,Isotropic>::CrystalStructure>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
86  break;
87  case Grain<dim>::W.Z:
88  CrossSlipModels<typename PeriodicElement<Grain<dim>::W.Z,Isotropic>::CrystalStructure>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
89  break;
90  default:
91  assert(0 && "DislocationCrossSlip: Material not implemented.");
92  break;
93  }
94 
95  }
96  }
97 
98  return crossSlipDeq;
99  }
