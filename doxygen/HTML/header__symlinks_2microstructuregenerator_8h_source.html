83  {
84 
85  const std::pair<LatticeVector<dim>,int> rp=randomPointInMesh();
86  const int& grainID=rp.second; // random grain ID
87  const LatticeVector<dim>& L0=rp.first; // random lattice position in the grain
88  const VectorDimD P0(L0.cartesian()); // cartesian position of L0
89  std::uniform_int_distribution<> distribution(0,poly.grain(grainID).slipSystems().size()-1);
90  const int rSS=distribution(generator); // a random SlipSystem ID
91  const SlipSystem& slipSystem(*poly.grain(grainID).slipSystems()[rSS]);
92  const VectorDimD b=slipSystem.s.cartesian(); // Burgers vector
93 
94 
95  VectorDimD n=slipSystem.unitNormal; // slip plane normal
96  std::uniform_real_distribution<> dis(0.0, 2.0*M_PI);
97  const double theta=dis(generator); // random angle of the dislocation line in the plane from screw orientation.
98  VectorDimD d=Eigen::AngleAxisd(theta, n)*b.normalized();
99 
100  bool isSessile=false;
101  if(sessileDensity/targetStraightDislocationDensity<fractionSessileStraightDislocationDensity)
102  {
103  n=b.normalized();
104  isSessile=true;
105  d=Eigen::AngleAxisd(theta, n)*n.cross(VectorDimD::Random()).normalized();
106  }
107 
108  const std::deque<VectorDimD> nodePos(DislocationInjector<dim>::straightLineBoundaryClosure(P0,d,n,grainID,mesh));
109 
110  const double lineLength=(nodePos.back()-nodePos.front()).norm();
111  // nodePos.push_back(nodePos[nodePos.size()-1]+1.0/3.0*(nodePos[0]-nodePos[nodePos.size()-1]));
112  // nodePos.push_back(nodePos[nodePos.size()-1]+2.0/3.0*(nodePos[0]-nodePos[nodePos.size()-1]));
113 
114  double dh=0.0;
115  if(enforceMonotonicHelicity)
116  {
117  dh=deltaHelicity(nodePos,b);
118  }
119 
120  // Write files
121  if( nodePos.size()>=3
122  && ((fabs(helicity+dh)>fabs(helicity) && fabs(dh)>FLT_EPSILON ) || helicity==0.0 || !enforceMonotonicHelicity))
123  {
124  // write node and edge file
125  for(size_t k=0;k<nodePos.size();++k)
126  {
127  configIO.nodes().emplace_back(nodeID+k,nodePos[k],Eigen::Matrix<double,1,3>::Zero(),1.0,snID,0);

132

135 
136  if(enforceMonotonicHelicity)

138  loopPoints.push_back(nodePos);
139  loopBurgers.push_back(b);
140  }
141 
142  loopID+=1;
143  snID+=1;

145  if(isSessile)
146  {
147  sessileDensity += lineLength/mesh.volume()/std::pow(poly.b_SI,2);

150  if(enforceMonotonicHelicity)
151  {
152  loopPoints.push_back(nodePos);
153  loopBurgers.push_back(b);
