52 void generalTheta(const double& theta,
53  const double& D,
54  const int& nSteps)
55 {
110  const double dt=1.0/nSteps;
111  const int SForder=1;
112  const int meshID=dim;
113 
114  SimplicialMesh<dim> mesh;
115  // Read the mesh files ./T/T_meshID.txt and ./N/N_meshID.txt
116  mesh.readMesh(meshID);
117 
118  // Create a FiniteElement object on the mesh
119  typedef LagrangeElement<dim,SForder> ElementType;
120  typedef FiniteElement<ElementType> FiniteElementType;
121  FiniteElementType fe(mesh);
122 
123  auto rho=fe.template trial<'r',1>();
124  rho=0.0;
125  SequentialOutputFile<'F',1> uFile;
126  uFile<<rho.onDomain();
127 
128  // Define integration domain
129  auto dV=fe.template domain<EntireDomain,3,GaussLegendre>();
130  auto lhs1=(test(rho),rho)*dV;
131  auto lhs2=(test(grad(rho)),theta*dt*D*grad(rho))*dV;
132 
133  auto rhs1=(test(rho),eval(rho))*dV;
134  auto rhs2=(test(grad(rho)),eval((1.0-theta)*dt*D*grad(rho)))*dV;
135  auto rhs3=(test(rho),MyFunction(dt))*dV;
136  auto weakProblem(lhs1+lhs2==rhs1-rhs2+rhs3); // weak problem
137 
138  const double solverTolerance=1.0e-6;
139  for (int k=0;k<nSteps;++k)
140  {
141  std::cout<<"general-theta STEP "<<k<<std::endl;
142  weakProblem.assemble();
143  rho=weakProblem.solve(solverTolerance);
144 
145  SequentialOutputFile<'F',1> uFile;
146  uFile<<rho.onDomain(); // Output function on domain
147  }
148 
149 }
